CREATE DATABASE IF NOT EXISTS crud;

USE crud;

CREATE TABLE persona (
  codigo INT NOT NULL AUTO_INCREMENT,
  nombre VARCHAR(100) NULL,
  apellido VARCHAR(100) NULL,
  edad INT NULL,
  numTelefono VARCHAR(20) NULL,
  numDni VARCHAR(20) NULL,
  PRIMARY KEY (codigo)
) ENGINE=InnoDB;
